YTD-AA-SDIs = 
VAR UltimoDia = 
	LASTNONBLANK (Calendario[Date];CALCULATE ( sum(Solicitudes[Nº SDIs Total]) ) )
VAR UltimoDiaAbsoluto = 
	LASTNONBLANK (
        Calendario[Date];
        CALCULATE ( sum(Solicitudes[Nº SDIs Total]); ALLEXCEPT ( Solicitudes; 'Calendario' ) ) )
VAR RangoActual =
	DATESBETWEEN ( Calendario[Date]; MIN ( Calendario[Date] ); UltimoDiaAbsoluto )
VAR RangoAnterior =
	//SAMEPERIODLASTYEAR ( RangoActual )
	DATESBETWEEN ( Calendario[Date]; DATEADD(FIRSTDATE(RangoActual);-1;YEAR); DATEADD(UltimoDiaAbsoluto;-1;YEAR) )
RETURN
	IF (
        	UltimoDiaAbsoluto >= MIN ( Calendario[Date] );
	        CALCULATE ( sum(Solicitudes[Nº SDIs Total]); RangoAnterior )
	)
